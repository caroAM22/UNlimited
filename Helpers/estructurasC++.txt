Vectores 

vector<int> v;
v.push_back(2);          // [2]
v[1] = 4;                // sets element at index 1 to 4 -> [2, 4, 7, 5]
v.erase(v.begin() + 1);  // removes element at index 1
v.erase(v.begin() + 1, v.begin() + 3);  // removes elements from index 1 to 2
v.erase(v.end() - 1);  // remove last position
